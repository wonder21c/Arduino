>### 2.실습에 대한 설명
```
앱인벤터로 피치값과 롤값을 나타내고 블루투스를 연결하는 앱을 만든다.
앱을 a bot과 연결하여 우리가 만든 맵에서 주행을 완료시킨다.
```
```c
#include <Servo.h>
Servo sr, sl;
void setup() {
  Serial.begin(9600);
  sr.attach(13); sl.attach(12);
}

void loop() {
  if (Serial.available()) {
    char b[2];  //pitch b[0], roll b[1]
    int p, r;
    Serial.readBytes(b, 2);
    p = b[0] * 3; 
    r = b[1];        
    sr.write(1500 + p - r);
    sl.write(1500 - p - r);
    Serial.write('1');
  }
}
```
